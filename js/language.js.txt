let currentLang = "en";
let translations = {};

async function setLanguage(lang) {
  currentLang = lang;
  const res = await fetch(`lang/${lang}.json`);
  translations = await res.json();
  applyLanguage();
}

function applyLanguage() {
  document.getElementById("title").textContent = translations.title;
  document.getElementById("cart-title").textContent = translations.cart;
  document.getElementById("total").textContent =
    `${translations.total}: $0`;
  document.getElementById("menu-upgrades").textContent =
    translations.menu.upgrades;
  document.getElementById("contact-title").textContent =
    translations.contact.title;
  document.getElementById("footer-text").textContent =
    translations.footer;
}

setLanguage("en");
